Add a list of meshes
====================

    import pyvista as pv

    from ansys.tools.visualization_interface import Plotter

    mesh1 = pv.Cube()
    mesh2 = pv.Sphere(center=(2, 0, 0))
    mesh_list = [mesh1, mesh2]
    # Create a plotter
    pl = Plotter()

    # Add a list of meshes to the plotter
    pl.plot(mesh_list)

    # Show the plotter
    pl.show()
